    <script>
        // Function to change content dynamically when button is clicked
        function changeContent() {
            document.getElementById("dynamic-text").innerHTML = "This is the new content after you clicked the button!";
        }

        // Form submission handling
        function submitForm(event) {
            event.preventDefault(); // Prevents the page from reloading on form submission
            const name = document.getElementById("name").value;
            const email = document.getElementById("email").value;

            // Basic validation
            if (name === "" || email === "") {
                document.getElementById("form-message").innerHTML = "Please fill in all fields.";
                document.getElementById("form-message").className = "error";
            } else {
                document.getElementById("form-message").innerHTML = `Thank you, ${name}! We will contact you at ${email}.`;
                document.getElementById("form-message").className = "message";
            }
        }
    </script>
